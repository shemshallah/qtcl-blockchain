-- Add density matrix hex + Mermin angles to quantum_snapshots
ALTER TABLE quantum_snapshots
  ADD COLUMN IF NOT EXISTS dm_hex            TEXT,
  ADD COLUMN IF NOT EXISTS mermin_angles     JSONB,
  ADD COLUMN IF NOT EXISTS dm_timestamp_ns   BIGINT,
  ADD COLUMN IF NOT EXISTS consensus_entropy DOUBLE PRECISION DEFAULT 0.0,
  ADD COLUMN IF NOT EXISTS w_state_strength  DOUBLE PRECISION DEFAULT 0.0,
  ADD COLUMN IF NOT EXISTS phase_coherence   DOUBLE PRECISION DEFAULT 0.0,
  ADD COLUMN IF NOT EXISTS entanglement_witness DOUBLE PRECISION DEFAULT 0.0,
  ADD COLUMN IF NOT EXISTS pq0_oracle        DOUBLE PRECISION DEFAULT 0.0,
  ADD COLUMN IF NOT EXISTS pq0_IV            DOUBLE PRECISION DEFAULT 0.0,
  ADD COLUMN IF NOT EXISTS pq0_V             DOUBLE PRECISION DEFAULT 0.0;

-- Index for client time-range queries
CREATE INDEX IF NOT EXISTS idx_qsnap_cycle
  ON quantum_snapshots (lattice_cycle DESC);

-- blocks table: ensure pq columns exist
ALTER TABLE blocks
  ADD COLUMN IF NOT EXISTS pq_curr BIGINT NOT NULL DEFAULT 0,
  ADD COLUMN IF NOT EXISTS pq_last BIGINT NOT NULL DEFAULT 0;

-- oracle_measurements: full per-cycle record
CREATE TABLE IF NOT EXISTS oracle_measurements (
  id               BIGSERIAL PRIMARY KEY,
  cycle            BIGINT NOT NULL,
  timestamp_ns     BIGINT NOT NULL,
  lattice_fidelity DOUBLE PRECISION,
  block_field_fidelity  DOUBLE PRECISION,
  block_field_coherence DOUBLE PRECISION,
  pq_curr          BIGINT DEFAULT 0,
  pq_last          BIGINT DEFAULT 0,
  oracle_count     INTEGER DEFAULT 5,
  fidelity_delta   DOUBLE PRECISION,
  window_type      TEXT,
  mermin_violation DOUBLE PRECISION,
  mermin_angles    JSONB,
  dm_hex           TEXT,
  per_node_json    JSONB
);
CREATE INDEX IF NOT EXISTS idx_om_cycle ON oracle_measurements (cycle DESC);
CREATE INDEX IF NOT EXISTS idx_om_ts    ON oracle_measurements (timestamp_ns DESC);
