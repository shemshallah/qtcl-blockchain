// ═══════════════════════════════════════════════════════════════════════════════
// QTCL EXPLORER ENGINE v4.0
// Transport: websocket (preferred) + polling (fallback) — matches server config
// Key normalization covers every snake/camel variant the server might emit
// Self-healing: HTTP fallback polls /metrics if WS drops
// Debug panel: toggle shows raw JSON of last payload + event log
// ═══════════════════════════════════════════════════════════════════════════════

const ORIGIN   = 'https://qtcl-blockchain.koyeb.app';
const API_BASE = 'https://qtcl-blockchain.koyeb.app';

// ── State ──────────────────────────────────────────────────────────────────────
let S = {
  blocks: [], mempool: [], miners: {},
  hrHist: [], diffHist: [],
  prevHeight: null, lastDataTs: null,
  reconnects: 0, httpActive: false, httpTimer: null,
  charts: { hr: null, diff: null },
  debugLog: [],
};

// ── Debug helpers ──────────────────────────────────────────────────────────────
function dbg(msg, color='#7a8db8') {
  const ts = new Date().toLocaleTimeString();
  S.debugLog.unshift(`<span style="color:${color}">[${ts}] ${msg}</span>`);
  if (S.debugLog.length > 80) S.debugLog.pop();
  const el = document.getElementById('debug-log');
  if (el) el.innerHTML = S.debugLog.join('');
}
function showPayload(d) {
  const el = document.getElementById('debug-payload');
  if (el) el.textContent = JSON.stringify(d, null, 2);
}
function toggleDebug() { document.getElementById('debug-panel').classList.toggle('visible'); }

// ── Socket.IO — matches server transports:['websocket','polling'] ──────────────
const socket = io(ORIGIN, {
  transports       : ['websocket', 'polling'],  // websocket first, polling fallback
  secure           : true,
  rejectUnauthorized: false,
  reconnection     : true,
  reconnectionDelay: 1500,
  reconnectionDelayMax: 12000,
  reconnectionAttempts: Infinity,
  timeout          : 20000,
});

// ── Connection lifecycle ───────────────────────────────────────────────────────
socket.on('connect', () => {
  const transport = socket.io.engine.transport.name.toUpperCase();
  dbg(`✅ Connected | id=${socket.id} | ${transport}`, '#00ff00');
  S.reconnects = 0;
  setStatus('live', `LIVE — ${transport}`);
  document.getElementById('transportBadge').textContent = transport;
  document.getElementById('reconnectInfo').textContent = '';
  stopHttp();
  // Pull fresh state immediately
  socket.emit('request_metrics');
  socket.emit('get_metrics');
});

socket.on('disconnect', reason => {
  dbg(`❌ Disconnected: ${reason}`, '#ff4444');
  setStatus('offline', `OFFLINE — ${reason}`);
  startHttp();
});

socket.on('connect_error', err => {
  S.reconnects++;
  dbg(`⚠️ connect_error #${S.reconnects}: ${err.message}`, '#ffaa00');
  setStatus('offline', `ERR — retry #${S.reconnects}`);
  document.getElementById('reconnectInfo').textContent = err.message;
  if (S.reconnects >= 3) startHttp();
});

socket.io.engine.on('upgrade', t => {
  dbg(`⬆️ Upgraded → ${t.name.toUpperCase()}`, '#00d9ff');
  document.getElementById('transportBadge').textContent = t.name.toUpperCase();
});

// ── Data events — catch every event name variant ──────────────────────────────
['metrics_update','update','state','blockchain_state','chain_update','metrics'].forEach(ev =>
  socket.on(ev, d => {
    dbg(`📊 ${ev} → keys:[${Object.keys(d||{}).slice(0,8).join(',')}…]`, '#00d9ff');
    showPayload(d);
    ingest(d);
  })
);

socket.on('new_block', b => {
  dbg(`📦 new_block h=${b?.height ?? b?.block_height ?? '?'}`, '#d4af37');
  const nb = normBlock(b);
  if (!S.blocks.find(x => x.height === nb.height)) {
    S.blocks.unshift(nb);
    if (S.blocks.length > 10) S.blocks.pop();
    renderBlocks(true);
  }
  // Also update height metric immediately
  if (nb.height) ingest({ block_height: nb.height });
  socket.emit('request_metrics');
});

socket.on('block', b => {
  const nb = normBlock(b);
  if (!S.blocks.find(x => x.height === nb.height)) {
    S.blocks.unshift(nb);
    if (S.blocks.length > 10) S.blocks.pop();
    renderBlocks(true);
  }
});

socket.on('mempool_update', d => {
  S.mempool = Array.isArray(d) ? d : (d?.transactions ?? d?.txs ?? []);
  renderMempool();
});

['miner_heartbeat','miner_update'].forEach(ev => socket.on(ev, d => {
  if (Array.isArray(d)) d.forEach(m => { if (m?.miner_id) S.miners[m.miner_id] = m; });
  else if (d?.miner_id) S.miners[d.miner_id] = d;
  renderMiners();
}));

// ── Core ingestion — normalizes EVERY field naming variant ─────────────────────
function ingest(raw) {
  if (!raw || typeof raw !== 'object') return;
  S.lastDataTs = Date.now();

  const pick = (...keys) => {
    for (const k of keys) if (raw[k] !== undefined && raw[k] !== null) return raw[k];
    return undefined;
  };

  // Block height
  const h = pick('block_height','height','blockHeight','chain_height','chainHeight');
  if (h !== undefined) {
    const n = parseInt(h, 10);
    if (!isNaN(n) && n >= 0) {
      if (S.prevHeight !== null && n > S.prevHeight) flashCard('card-height', '+' + (n - S.prevHeight));
      S.prevHeight = n;
      setText('height', n.toLocaleString());
    }
  }

  // Difficulty
  const d = pick('difficulty','network_difficulty','diff');
  if (d !== undefined) {
    const n = parseFloat(d);
    if (!isNaN(n)) {
      setText('difficulty', n > 1e6 ? (n/1e6).toFixed(2)+'M' : n > 1e3 ? (n/1e3).toFixed(2)+'K' : n.toFixed(2));
      S.diffHist.push({ t: new Date(), v: n });
      if (S.diffHist.length > 1440) S.diffHist.shift();
    }
  }

  // Hash rate
  const hr = pick('network_hash_rate','hash_rate','hashrate','hashRate');
  if (hr !== undefined) {
    const n = parseFloat(hr);
    if (!isNaN(n)) {
      const f = fmtHR(n);
      setText('hashRate', f.v);
      document.getElementById('hashRate-unit').textContent = f.u;
      S.hrHist.push({ t: new Date(), v: n });
      if (S.hrHist.length > 1440) S.hrHist.shift();
    }
  }

  // W-state fidelity (0–1 float OR 0–100 float)
  const fid = pick('w_state_fidelity','fidelity','wstate_fidelity','quantum_fidelity');
  if (fid !== undefined) {
    const n = parseFloat(fid);
    if (!isNaN(n)) setText('fidelity', (n > 1 ? n : n * 100).toFixed(2));
  }

  // Peers
  const pc = pick('peer_count','peers','peerCount','active_peers');
  if (pc !== undefined) { const n = parseInt(pc, 10); if (!isNaN(n)) setText('peers', n); }

  // Mempool size
  const mp = pick('mempool_size','pending_count','mempool_count');
  if (mp !== undefined) {
    const n = parseInt(mp, 10);
    if (!isNaN(n)) {
      setText('mempool', n);
      document.getElementById('mempool-count').textContent = n;
    }
  }

  // Last block time ago (unix timestamp OR seconds-ago value)
  const lbt = pick('last_block_time_ago','time_since_block','last_block_time');
  if (lbt !== undefined) {
    let s = 0;
    if (typeof lbt === 'number') s = lbt > 1e9 ? Math.floor(Date.now()/1000 - lbt) : lbt;
    else if (typeof lbt === 'string') {
      const p = Date.parse(lbt);
      s = isNaN(p) ? parseFloat(lbt) || 0 : Math.floor((Date.now() - p) / 1000);
    }
    setText('lastBlockTime', s >= 0 ? s.toFixed(0) : '—');
  }

  // Active miners count (from registry or direct field)
  const am = pick('active_miners','miner_count');
  if (am !== undefined) { const n = parseInt(am, 10); if (!isNaN(n)) setText('activeMiners', n); }
  else { const mc = Object.keys(S.miners).length; if (mc > 0) setText('activeMiners', mc); }

  // Embedded recent_blocks array
  const rb = raw.recent_blocks ?? raw.blocks ?? raw.recentBlocks;
  if (Array.isArray(rb) && rb.length) {
    rb.forEach(b => {
      const nb = normBlock(b);
      if (!S.blocks.find(x => x.height === nb.height)) S.blocks.unshift(nb);
    });
    S.blocks.sort((a, b) => b.height - a.height);
    S.blocks = S.blocks.slice(0, 10);
    renderBlocks(false);
  }

  // Embedded mempool transactions
  const mt = raw.mempool_txs ?? raw.mempool ?? raw.pending_transactions;
  if (Array.isArray(mt) && mt.length) { S.mempool = mt; renderMempool(); }

  // Embedded miners
  const mi = raw.miners;
  if (mi && typeof mi === 'object' && !Array.isArray(mi)) {
    Object.assign(S.miners, mi);
    renderMiners();
  }

  updateCharts();
}

// ── Block normalization ────────────────────────────────────────────────────────
function normBlock(b) {
  if (!b) return {};
  return {
    height   : b.height ?? b.block_height ?? b.blockHeight ?? 0,
    hash     : b.hash ?? b.block_hash ?? b.id ?? '',
    miner    : b.miner ?? b.miner_id ?? b.validator_public_key ?? b.mined_by ?? 'unknown',
    timestamp: b.timestamp ?? b.time ?? b.created_at ?? (Date.now()/1000),
    tx_count : b.tx_count ?? b.transactions ?? b.txCount ?? 0,
    entropy  : b.entropy ?? b.entropy_score ?? null,
  };
}

// ── DOM setters ────────────────────────────────────────────────────────────────
function setText(id, val) {
  const e = document.getElementById(id);
  if (!e) return;
  e.textContent = val;
  e.classList.remove('stale');
}
function flashCard(id, delta) {
  const c = document.getElementById(id); if (!c) return;
  c.classList.remove('updated'); void c.offsetWidth; c.classList.add('updated');
  if (delta !== undefined) {
    const de = c.querySelector('.metric-delta');
    if (de) { de.textContent = delta; de.className = 'metric-delta' + (String(delta).startsWith('-') ? ' neg' : ''); }
  }
}
function setStatus(s, txt) {
  const dot = document.getElementById('statusDot'), t = document.getElementById('statusText');
  if (dot) dot.className = 'status-dot' + (s==='live'?' live':'');
  if (t) { t.className='status-text'+(s==='live'?' live':''); t.textContent=txt; }
}

// ── Renderers ──────────────────────────────────────────────────────────────────
function renderBlocks(animFirst) {
  const c = document.getElementById('blocks-list');
  document.getElementById('block-count').textContent = S.blocks.length;
  if (!S.blocks.length) { c.innerHTML='<div style="padding:20px;color:#7a8db8">No blocks yet</div>'; return; }
  c.innerHTML = S.blocks.map((b, i) => `
    <div class="block-item ${animFirst&&i===0?'fresh':''}">
      <div class="block-height">Block #${Number(b.height).toLocaleString()}</div>
      <div class="block-field">Hash: <span class="block-field-value">${String(b.hash||'').substring(0,20)}…</span></div>
      <div class="block-field">Miner: <span class="block-field-value">${String(b.miner||'unknown').substring(0,18)}…</span></div>
      <div class="block-field">Time: <span class="block-field-value">${new Date(b.timestamp>1e10?b.timestamp:b.timestamp*1000).toLocaleString()}</span></div>
      <div class="block-field">Txns: <span class="block-field-value">${b.tx_count}</span></div>
      ${b.entropy!=null?`<div class="block-field">Entropy: <span class="block-field-value">${typeof b.entropy==='number'?b.entropy.toFixed(6):b.entropy}</span></div>`:''}
    </div>`).join('');
}

function renderMempool() {
  const tbody = document.getElementById('mempool-list');
  document.getElementById('mempool-count').textContent = S.mempool.length;
  if (!S.mempool.length) { tbody.innerHTML='<tr><td colspan="6" style="text-align:center;color:#7a8db8">No pending transactions</td></tr>'; return; }
  tbody.innerHTML = S.mempool.slice(0,10).map(tx => `<tr>
    <td>${String(tx.hash??tx.tx_hash??tx.id??'').substring(0,16)}</td>
    <td>${String(tx.from??tx.from_address??tx.sender??'').substring(0,12)}…</td>
    <td>${String(tx.to??tx.to_address??tx.receiver??'').substring(0,12)}…</td>
    <td>${tx.amount??0}</td><td>${tx.fee??0}</td>
    <td style="color:#00ff00">Pending</td></tr>`).join('');
}

function renderMiners() {
  const tbody = document.getElementById('miners-list');
  const arr = Object.values(S.miners)
    .sort((a,b)=>(b.blocks_mined??b.block_count??0)-(a.blocks_mined??a.block_count??0)).slice(0,10);
  document.getElementById('miner-count').textContent = arr.length;
  if (arr.length > 0) setText('activeMiners', arr.length);
  if (!arr.length) { tbody.innerHTML='<tr><td colspan="6" style="text-align:center;color:#7a8db8">No miner data yet</td></tr>'; return; }
  tbody.innerHTML = arr.map((m,i) => {
    const f = fmtHR(m.hash_rate ?? m.hashrate ?? 0);
    const ts = m.last_block_time ?? m.last_block_ts ?? m.last_seen ?? Date.now()/1000;
    return `<tr>
      <td>${i+1}</td><td>${String(m.miner_id??m.id??'?').substring(0,18)}…</td>
      <td>${m.blocks_mined??m.block_count??0}</td>
      <td>${f.v} ${f.u}</td>
      <td>${new Date(ts>1e10?ts:ts*1000).toLocaleString()}</td>
      <td>${String(m.wallet_address??m.wallet??m.address??m.miner_id??'').substring(0,12)}…</td></tr>`;
  }).join('');
}

// ── HTTP fallback polling ──────────────────────────────────────────────────────
function startHttp() {
  if (S.httpActive) return;
  S.httpActive = true;
  dbg('🔄 HTTP fallback active', '#ffaa00');
  httpPoll();
  S.httpTimer = setInterval(httpPoll, 5000);
}
function stopHttp() {
  if (!S.httpActive) return;
  S.httpActive = false;
  clearInterval(S.httpTimer);
  dbg('✅ HTTP fallback stopped', '#00ff00');
}
async function httpPoll() {
  const eps = ['/metrics', '/api/metrics', '/stats', '/api/stats', '/api/health'];
  for (const ep of eps) {
    try {
      const r = await fetch(API_BASE + ep, { mode:'cors', signal: AbortSignal.timeout(4000) });
      if (r.ok) {
        const d = await r.json();
        dbg(`📡 HTTP ${ep} → ${Object.keys(d).slice(0,4).join(',')}…`, '#00d9ff');
        ingest(d); showPayload(d); return;
      }
    } catch(e) { /* try next */ }
  }
}

// ── Format hash rate ───────────────────────────────────────────────────────────
function fmtHR(n) {
  n = parseFloat(n) || 0;
  if (n>=1e12) return {v:(n/1e12).toFixed(2),u:'TH/s'};
  if (n>=1e9)  return {v:(n/1e9).toFixed(2), u:'GH/s'};
  if (n>=1e6)  return {v:(n/1e6).toFixed(2), u:'MH/s'};
  if (n>=1e3)  return {v:(n/1e3).toFixed(2), u:'KH/s'};
  return {v:n.toFixed(0), u:'H/s'};
}
